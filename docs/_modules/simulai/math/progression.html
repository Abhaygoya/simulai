# Geometrical progression

[docs]def gp(init, factor, n) -> list:
    if n == 1:
        return [init]
    elif n > 1:
        return gp(init, factor, n - 1) + [factor * gp(init, factor, n - 1)[-1]]
    else:
        raise Exception("n must be higher than 0")
